Returns the parsed environment variable value.

Arguments

  • key - The environment variable name

Example

fn main() {
    envmnt::set("ENV_VAR", "123");
    let value: i32 = envmnt::get_parse("ENV_VAR").unwrap();
    assert_eq!(value, 123);
}
